name: Sync external repo to feature/new-updates

on:
  schedule:
    - cron: '0 2 * * *' # каждый день в 2:00
  workflow_dispatch:     # ручной запуск

jobs:
  sync:
    runs-on: ubuntu-latest

    steps:
      - name: Checkout corporate repo
        uses: actions/checkout@v3
        with:
          fetch-depth: 0

      - name: Configure Git
        run: |
          git config user.name "github-actions"
          git config user.email "github-actions@github.com"

      - name: Add upstream remote
        run: |
          git remote add upstream https://github.com/rosetta-models/blank-base-model.git || true
          git fetch upstream

      - name: Update feature/new-updates branch
        run: |
          git checkout feature/new-updates
          git reset --hard upstream/main

      - name: Check for changes before push
        id: changes
        run: |
          git fetch origin feature/new-updates
          if git diff --quiet feature/new-updates origin/feature/new-updates; then
            echo "no_changes=true" >> $GITHUB_ENV
          else
            echo "no_changes=false" >> $GITHUB_ENV
          fi

      - name: Push changes if any
        if: env.no_changes == 'false'
        env:
          GH_TOKEN: ${{ secrets.GITHUB_TOKEN }}
        run: |
          git push -f https://x-access-token:${GH_TOKEN}@github.com/<org>/company-blank-base-model.git feature/new-updates
